  async seed() {
    let ROLES_ROWS = 0;
    let STATE_ROWS = 0;
    let COUNTRY_ROWS = 0;
    let LANGUAGE_ROWS = 0;
    let SUSCRIPTION_ROWS = 0;
    let PLATFORM_ROWS = 0;
    let THIRD_PARTY_INTEREST_ROWS = 0;
    let POINTS_CONVERSION_ROWS = 0;
    let BANK_ROWS = 0;

    await this.role()
      .then(completed => {
        ROLES_ROWS = completed;
        return ROLES_ROWS;
      })
      .catch(error => {
        return Promise.reject(error);
      });

    await this.state()
      .then(completed => {
        STATE_ROWS = completed;
